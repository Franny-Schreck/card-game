{NAME} "bureaucracy"
{PLAY-COST} 2
{SHOP-COST} 4

{DISPLAYNAME} "Bureaucracy"

{DESCRIPTION} "Draw 1 card and gain 3 governance points. When played, creates a guild center or government building."

{IMAGE} "bureaucracy"

{TAGS} "shop"

{FREQUENCY} 50

{TAGS} "shop"

{TARGET} "local"

{CONDITION}
	(cond (!= %max-bureaucracy-level %bureaucracy-level) "There is no more space for bureaucracy here")

{EFFECT}
	(if (< (card-count "draw") 1)
		(put-card "draw" (take-card -1 "discard" "random"))
		())
	(put-card "hand" (take-card 1 "draw" "top"))
	(setf + $gp 3)
	(setf + %bureaucracy-level 1)
