{NAME} "farm"
{PLAY-COST} 1
{SHOP-COST} 2

{DISPLAYNAME} "Agriculture"

{DESCRIPTION} "Gain 3 governance point and 3 contentment, lose 1 florin."

{IMAGE} "agriculture"

{TAGS} "shop"

{FREQUENCY} 80

{TARGET} "local"

{CONDITION}
	(cond (>= $fl 1) "Need at least two {fl}")
	(cond (!= %max-farm-level %farm-level) "There is no more space for fields here")

{EFFECT}
	(setf - $fl 1)
	(setf + $gp 3)
	(setf + %contentment 3)
	(setf + %farm-level 1)
