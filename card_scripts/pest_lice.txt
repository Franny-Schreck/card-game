{NAME} "pest_lice"
{PLAY-COST} -1
{SHOP-COST} 0

{DISPLAYNAME} "LICE"

{DESCRIPTION} "Some sickness is spreading through lice, many people get sick. Lose 3 Florin and every district 2 contentment"

{IMAGE} "pest-lice"

{TAGS} "sticky"

{BACKGROUND} "pest_base"

{FREQUENCY} 80

{TARGET} "global"

{USES} "0"

{CONDITION} false

{EFFECT} ()

{ON-TURN-END}
	(if (!= (card-attr self "age") 0)
		(each
			(animate-play self)
			(put-card "void" self)
			(put-card "hand" (take-card 1 "card:pest_dies" "top"))
			(all-districts (setf - %contentment 2))
			(setf - $fl 3))
		())
