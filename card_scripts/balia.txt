{NAME} "balia"
{PLAY-COST} 1
{SHOP-COST} 2

{DISPLAYNAME} "Balia"

{DESCRIPTION} "Draw 1 card, gain 3 government points, and lose 2 contentment"

{IMAGE} "balia"

{TAGS} "shop"

{TARGET} "local"

{FREQUENCY} 50

{CONDITION} true

{EFFECT}
	(if (< (card-count "draw") 1)
		(put-card "draw" (take-card -1 "discard" "random"))
		())
	(put-card "hand" (take-card 1 "draw" "top"))
	(setf + $gp 3)
	(setf - %contentment 2)
