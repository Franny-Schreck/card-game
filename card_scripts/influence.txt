{NAME} "influence"
{PLAY-COST} 4
{SHOP-COST} 4

{DISPLAYNAME} "Influence"

{DESCRIPTION} "Lose 7 Florin. Gain 8 Governance Points and 3 contentment."

{IMAGE} "influence"

{TAGS} "shop"

{FREQUENCY} 70

{TARGET} "local"

{CONDITION}
	(>= $fl 7)
	(cond (!= %max-palazzo-level %palazzo-level) "There is no more space for fancy palazzos here")

{EFFECT}
	(setf - $fl 7)
	(setf + $gp 8)
	(setf + %contentment 3)
	(setf + %palazzo-level 1)
