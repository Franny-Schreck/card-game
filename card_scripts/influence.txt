{NAME} "influence"
{PLAY-COST} 4
{SHOP-COST} 5

{DISPLAYNAME} "Influence"

{DESCRIPTION} "Lose 10 fl. Gain 2 gp and 1 contentment."

{IMAGE} "influence"

{TAGS} "shop"

{FREQUENCY} 70

{TARGET} "local"

{CONDITION}
	(>= $fl 10)
	(cond (!= %max-palazzo-level %palazzo-level) "There is no more space for fancy palazzos here")

{EFFECT}
	(setf - $fl 10)
	(setf + $gp 3)
	(setf + %contentment 1)
	(setf + %palazzo-level 1)
